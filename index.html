Tutorial

1 $ export GITHUB_USERNAME=<имя_пользователя>
2 $ export GITHUB_EMAIL=<адрес_почтового_ящика>
3 $ alias edit=<nano|vi|vim|subl>
1 $ cd ${GITHUB_USERNAME}/workspace
2 $ source scripts/activate
1 $ mkdir projects/lab03 && cd projects/lab03
2 $ git init
3 $ git config --global user.name ${GITHUB_USERNAME}
4 $ git config --global user.email ${GITHUB_EMAIL}
5 $ git config -e --global
6 $ git remote add origin https://github.com/${GITHUB_USERNAME}/lab03.git
7 $ git pull origin master
8 $ touch README.md
9 $ git status
10 $ git add README.md
11 $ git commit -m"added README.md"
12 $ git push origin master
